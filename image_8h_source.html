49#define IMAGE_OCT__(xr, yr, zr, l1, l2, l3, ra, rb, rc) \
50 { \
51 real fx = xr * ra.x; \
52 real fy = yr * ra.x; \
53 real fz = zr * ra.x; \
54 fx -= REAL_FLOOR(0.5f + fx); \
55 fy -= REAL_FLOOR(0.5f + fy); \
56 fz -= REAL_FLOOR(0.5f + fz); \

60 fz -= REAL_SIGN(0.5f, fz); \
