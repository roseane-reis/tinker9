149 real zi = z[i];

151 int ix = zaxis[i].xaxis;
152 int iy = INT_ABS(zaxis[i].yaxis) - 1;

154 // the default identity matrix

156
157 if (polaxe != LFRM_NONE) {
158 real* restrict xx = &a[0][0];
159 real* restrict yy = &a[1][0];
160 real* restrict zz = &a[2][0];
161
162 // STEP 1: PICK Z AND NORM Z
163 // pick z
164 zz[0] = x[iz] - xi;
165 zz[1] = y[iz] - yi;
166 zz[2] = z[iz] - zi;
167 // norm z
168 rotpoleNorm(zz);
169
170 // STEP 2: PICK X AND NORM X
171 // even if it is not needef for z then x)
172 if (polaxe == LFRM_Z_ONLY) {
173 // pick x
174 int okay = !(REAL_ABS(zz[0]) > 0.866f);
175 xx[0] = (okay ? 1 : 0);
176 xx[1] = (okay ? 0 : 1);
177 xx[2] = 0;
178 } else {
179 // pick x
180 xx[0] = x[ix] - xi;
181 xx[1] = y[ix] - yi;
182 xx[2] = z[ix] - zi;
183 rotpoleNorm(xx);
184 }
185
186 // STEP 3: PICK Y AND NORM Y
187 // only for z biscector and 3 fold
188 if (polaxe == LFRM_Z_BISECT || polaxe == LFRM_3_FOLD) {
189 yy[0] = x[iy] - xi;
190 yy[1] = y[iy] - yi;
191 yy[2] = z[iy] - zi;
192 rotpoleNorm(yy);
193 }
194

196 if (polaxe == LFRM_BISECTOR) {
197 rotpoleAddBy(zz, xx);
198 rotpoleNorm(zz);
199 } else if (polaxe == LFRM_Z_BISECT) {
200 rotpoleAddBy(xx, yy);
201 rotpoleNorm(xx);
202 } else if (polaxe == LFRM_3_FOLD) {
203 rotpoleAddBy2(zz, xx, yy);
204 rotpoleNorm(zz);
205 }
206
207 // STEP 5

209 real dotxz = xx[0] * zz[0] + xx[1] * zz[1] + xx[2] * zz[2];
210 xx[0] -= dotxz * zz[0];
211 xx[1] -= dotxz * zz[1];
212 xx[2] -= dotxz * zz[2];
213 // norm x
214 rotpoleNorm(xx);
215 // y = z cross x
216 a[1][0] = a[0][2] * a[2][1] - a[0][1] * a[2][2];
217 a[1][1] = a[0][0] * a[2][2] - a[0][2] * a[2][0];
218 a[1][2] = a[0][1] * a[2][0] - a[0][0] * a[2][1];
